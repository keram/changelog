Hello Humans!

This is simple guide how you can use this project.

You can use my default list of projects or you can build your own!
Just create file "source.json" in format below and change url to:
http://localhost/work/keramovo/changelog/html/?source=url_to_your_source.json

Example format of source.json

{ "author" : "@marek_public", "title" : "Dev Changelog", "projects" : [ { "title" : "Ruby on Rails", "url" : "http://weblog.rubyonrails.org/", "iframeable" : 1, "web" : "http://rubyonrails.org/", "guide" : "http://guides.rubyonrails.org/", "download" : "http://rubyonrails.org/download", "repository" : "https://github.com/rails/rails/", "branches" : [ { "title" : "master", "url" : "https://github.com/rails/rails/commits/master" }, { "title" : "3-2-stable", "url" : "https://github.com/rails/rails/commits/3-2-stable" } ] }, { "title" : "Node.js", "url" : "http://nodejs.org/changelog.html", "iframeable" : 1, "web" : "http://nodejs.org/", "guide" : "http://guides.rubyonrails.org/", "download" : "http://rubyonrails.org/download", "repository" : "https://github.com/joyent/node/", "branches" : [ { "title" : "master", "url" : "https://github.com/joyent/node/commits/master" } ] } ] }
